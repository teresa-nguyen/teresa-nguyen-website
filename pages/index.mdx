import { Carousel } from '@mantine/carousel'
import { Card, Image, Modal, UnstyledButton } from '@mantine/core'
import { useState } from 'react'

export const ClickableImage = ({ src, alt }) => {
  const [opened, setOpened] = useState(false)
  return (
    <>
      <Modal opened={opened} onClose={() => setOpened(false)} size={800}>
        <Image src={src} alt={alt} />
      </Modal>
      <UnstyledButton onClick={() => setOpened(true)}>
        <Image src={src} alt={alt} fit="scale-down" height={200} />
      </UnstyledButton>
    </>
  )
}
